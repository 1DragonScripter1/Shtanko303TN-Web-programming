Програмування

Ось приклад коду на C++, який я вивчаю:

            
#include <iostream>

using namespace std;

int main() {
  int n;
  cout << "Enter a positive integer: ";
  cin >> n;

  long long factorial = 1;

  if (n < 0) {
    cout << "Factorial is not defined for negative numbers." << endl;
  } else {
    for (int i = 1; i <= n; ++i) {
      factorial *= i;
    }
    cout << "Factorial of " << n << " = " << factorial << endl;
  }
  return 0;
}
